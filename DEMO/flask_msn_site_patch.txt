@app.route("/recommendations", methods=["GET", "POST"])
def recommendations():
    u = current_user()
    if not u:
        flash("Please sign in to get personalized AI recommendations.", "error")
        return redirect(url_for("login"))

    repo = get_repo()
    history_items = read_history_list(200)

    model_key = selected_model_key()
    selected_label = model_label(model_key)
    service = get_model_service(ROOT, model_key)

    mind_raw = u.get("mind_user_id")
    hist = repo.user_history_ids(mind_raw) if mind_raw else set()
    session_history = set(session.get("read_ids", []))
    all_history = list(hist | session_history)

    recommendations = []
    generation_time = None
    generated = False
    ranking_note = service.status_message() if hasattr(service, "status_message") else ""
    max_candidates = 100

    if request.method == "POST" and request.form.get("action") == "generate":
        generated = True
        session["selected_model"] = model_key

        if not all_history:
            flash("No reading history found. Please read some articles first to get recommendations.", "warning")
        elif not service.available:
            flash(f"{selected_label} is not available: {getattr(service, 'load_error', 'Unknown error')}", "error")
        else:
            try:
                start_time = time.time()
                neural_recs = service.recommend_trained(
                    history_news_ids=all_history,
                    k=12,
                    max_history=50,
                )
                generation_time = time.time() - start_time

                for nid, score in neural_recs:
                    rec_dict = article_dict(repo, nid, score, f"AI · {selected_label}")
                    if rec_dict:
                        recommendations.append(rec_dict)

                if recommendations:
                    flash(f"✅ Generated {len(recommendations)} recommendations by {selected_label} in {generation_time:.2f}s!", "success")
                    ranking_note = service.status_message()
                else:
                    flash("No recommendations generated. Try reading more articles or another MIND user ID.", "warning")
            except Exception as e:
                flash(f"Error generating recommendations: {str(e)}", "error")
                ranking_note = f"Error: {str(e)}"

    return render_template(
        "msn/recommendations.html",
        user=u,
        history_items=history_items,
        recommendations=recommendations,
        generated=generated,
        generation_time=generation_time,
        ranking_note=ranking_note,
        max_candidates=max_candidates,
        model_options=model_options(ROOT),
        selected_model=model_key,
        selected_model_label=selected_label,
    )
